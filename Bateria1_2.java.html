package org.example;

import java.util.Scanner;

public class Main {
    static void main() {

        final int JORNADA_SIN_EXTRAS = 40;
        final double PLUS_TARIFA_EXTRA = 1.5;

        Scanner teclado = new Scanner(System.in);

        System.out.println("Introduzca las horas");
        int horas = teclado.nextInt();

        System.out.println("Introduzca la tarifa por horas");
        double tarifa = teclado.nextDouble();

        if (horas>JORNADA_SIN_EXTRAS){

            double extras = horas - JORNADA_SIN_EXTRAS;
            double tarifa_extras = tarifa * PLUS_TARIFA_EXTRA;
            double salario_extra = extras*tarifa_extras;

            System.out.println("Salario (con extras) es: " + (salario_extra + (JORNADA_SIN_EXTRAS*tarifa)));

        }else{
            double salario = horas * tarifa;
            System.out.println("El salario bruto es " + salario);
        }


    }
}
